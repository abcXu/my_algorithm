### 一、哈希表概念

* **哈希表是**：一种能够根据键值直接确定数据存放位置的数据结构

* **哈希函数**：键到值的映射法则

* **造表**：利用哈希函数构建哈希表

* **冲突**不同的键值得到同一哈希值

* **装填因子**键值数量与哈希表长度的比值

#### 1、有效的字母异位词

[题目链接](https://leetcode.cn/problems/valid-anagram/)

```cpp
class Solution {
public:
    bool isAnagram(string s, string t) {
        int record[26] = {0};   // 对应26个字母
        for(int i = 0;i<s.length();i++){
            record[s[i]-'a']++;    // 记录s中每个字母出现的频次
        }
        for(int i = 0;i<t.length();i++){
            record[t[i]-'a']--; 
        }
        for(int i = 0;i<26;i++){
            if(record[i]!=0) return false;  // 满足题目要求则record所有元素值为0
        }
        return true;
    }
};
```

- 时间复杂度: O(n)
- 空间复杂度: O(1)

#### 2、两个数组的交集

[题目链接](https://leetcode.cn/problems/intersection-of-two-arrays/)

```cpp
class Solution {
public:
    vector<int> intersection(vector<int>& nums1, vector<int>& nums2) {
        unordered_set<int> result_set; // 存放结果，之所以用set是为了给结果集去重
        unordered_set<int> nums_set(nums1.begin(), nums1.end());
        for (int num : nums2) {
            // 发现nums2的元素 在nums_set里又出现过
            if (nums_set.find(num) != nums_set.end()) {
                result_set.insert(num);
            }
        }
        return vector<int>(result_set.begin(), result_set.end());
    }
};
```

- 时间复杂度: O(mn)
- 空间复杂度: O(n)

#### 3、快乐数

[题目链接](https://leetcode.cn/problems/happy-number/)

* 抓住题目给定信息**非快乐数将会出现无限循环**
* 只需要判断计算的过程中是否重复出现某个数字，即循环开始了
* 可以利用`undered_set`的特性判断是否重复出现某个值

```cpp
class Solution {
public:
    // 用于获取每个数字每一位之和
    int getSum(int n){  
        int sum = 0;
        while(n){
            sum += (n%10)*(n%10);
            n/=10;
        }
        return sum;
    }
    bool isHappy(int n) {
        unordered_set<int> set;
        while(true){
            int sum = getSum(n);
            if(sum==1) return true;

            if(set.find(sum)!=set.end())    // 再次出现，循环了
            {
                return false;
            }else{  // 没有出现，加入集合
                set.insert(sum);
            }
            n = sum;
        }
    }
};
```

#### 4、N数之和

**两数之和**

​	[题目链接](https://leetcode.cn/problems/two-sum/)

* 利用unordered_map加上内置的find函数

```cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        unordered_map<int,int> m;
        for(int i=0;i<nums.size();i++)
        {     
            if(m.find(target-nums[i])!=m.end())	// 查到
            {
                return{m[target-nums[i]],i};
            }
            m[nums[i]]=i;      
        }   
        return {};
    }
};
```



**三数之和**

​	[题目链接](https://leetcode.cn/problems/3sum/description/)

* 先将数组进行一次排列
* 定义指针变量`i , j , k`
* `i `从0开始，`j = i +1`, ` k `从后往前移
* 比较三数之和与0的大小 ， 小于， ` j`往右移，大于则`k `往左移，等于则将三数加入结果集合
* `k--,j++`进行下一轮迭代

```cpp
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        sort(nums.begin(), nums.end()); // 排序
        vector<vector<int>> ans;
        int n = nums.size();
        for(int i = 0;i<n-2&&nums[i]<=0;i++)
        {
            if(i&&nums[i]==nums[i-1])
            {
                continue;
            }
            int j = i+1,k = n-1;
            while(j<k){
                int x = nums[i]+nums[j]+nums[k];
                if(x<0) // 负的值过大，调小负的值使值趋于0 不可调大整数，因为k的值是从后边递减的
                {
                    ++j;

                }else if(x>0){
                    --k;
                }else{  // x==0的情况
                    ans.push_back({nums[i],nums[j++],nums[k--]});   // 注意此处的后置运算符
                    while(j<k&&nums[j]==nums[j-1]){ // 当前j是否和前一个值相等，相等则不考虑
                        ++j;
                    }
                    while(j<k&&nums[k]==nums[k+1]){ // 同上
                        --k;
                    }
                }
            }
        }
        return ans;
    }
};
```
